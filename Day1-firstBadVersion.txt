bool isBadVersion(int version);
 
class Solution {
public:
    int firstBadVersion(int n) {
        int i = 1;
        int j = n;
        int k = (i + (j - i) / 2);
        while (i < j) {
            if (isBadVersion(k)) {
                j = k - 1; 
            } else {
                i = k + 1; 
            }
            k = (i + (j - i) / 2); 
        }
        return isBadVersion(i) ? i : i + 1;
    }
};
